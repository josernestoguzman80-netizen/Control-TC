<script>
    let todasLasCompras = [];
    const CORTE = 15;
    const PAGO = 10;
    const TASA_MENSUAL = 0.05;

    // FUNCIÓN PARA GUARDAR EN EL NAVEGADOR
    function guardarDatos() {
        localStorage.setItem('datosComprasGuzman', JSON.stringify(todasLasCompras));
    }

    // FUNCIÓN PARA CARGAR AL ABRIR LA PÁGINA
    function cargarDatos() {
        const datosGuardados = localStorage.getItem('datosComprasGuzman');
        if (datosGuardados) {
            todasLasCompras = JSON.parse(datosGuardados);
            render();
        }
    }

    function registrar() {
        const d = document.getElementById('desc').value;
        const m = parseFloat(document.getElementById('monto').value);
        const fTxt = document.getElementById('fechaCompra').value;
        if (!d || isNaN(m) || !fTxt) return alert("Llena todos los campos.");

        const partes = fTxt.split('-');
        const f = new Date(partes[0], partes[1]-1, partes[2]);
        let fLimite = new Date(f);
        if (f.getDate() > CORTE) fLimite.setMonth(fLimite.getMonth() + 2);
        else fLimite.setMonth(fLimite.getMonth() + 1);
        fLimite.setDate(PAGO);

        todasLasCompras.push({
            id: Date.now(), desc: d, monto: m, pagado: false,
            fechaStr: f.toLocaleDateString('es-GT'),
            fechaLimiteStr: fLimite.toLocaleDateString('es-GT'),
            mesSort: fLimite.getFullYear() * 100 + (fLimite.getMonth() + 1),
            nombreCiclo: fLimite.toLocaleDateString('es-GT', { month: 'long', year: 'numeric' })
        });
        document.getElementById('desc').value = ""; document.getElementById('monto').value = "";
        guardarDatos(); // Guardamos después de registrar
        render();
    }

    function toggle(id) {
        const i = todasLasCompras.findIndex(x => x.id === id);
        todasLasCompras[i].pagado = !todasLasCompras[i].pagado;
        guardarDatos(); // Guardamos después de cambiar el estado
        render();
    }

    function pagarTodoElMes(mesSort) {
        todasLasCompras.forEach(c => {
            if (c.mesSort == mesSort) c.pagado = true;
        });
        guardarDatos(); // Guardamos después de pagar todo
        render();
    }

    function borrar(id) {
        todasLasCompras = todasLasCompras.filter(x => x.id !== id);
        guardarDatos(); // Guardamos después de borrar
        render();
    }

    function render() {
        const cont = document.getElementById('contenedorCiclos');
        cont.innerHTML = "";
        let capPendTotal = 0;

        const grupos = {};
        todasLasCompras.forEach(c => {
            if (!grupos[c.mesSort]) grupos[c.mesSort] = { nombre: c.nombreCiclo, limite: c.fechaLimiteStr, items: [] };
            grupos[c.mesSort].items.push(c);
        });

        Object.keys(grupos).sort((a,b) => a-b).forEach(k => {
            const ciclo = grupos[k];
            let totalPendienteCiclo = 0;
            ciclo.items.forEach(item => {
                if (!item.pagado) {
                    totalPendienteCiclo += item.monto;
                    capPendTotal += item.monto;
                }
            });

            let html = `
                <div class="ciclo-card">
                    <div class="ciclo-header">
                        <span>PAGO DE ${ciclo.nombre.toUpperCase()}</span>
                        <button class="btn-pagar-todo" onclick="pagarTodoElMes(${k})">Marcar todo como pagado</button>
                    </div>
                    <table>
                        <thead><tr><th>¿Pagado?</th><th>Fecha</th><th>Descripción</th><th>Monto</th><th>X</th></tr></thead>
                        <tbody>`;

            ciclo.items.forEach(item => {
                html += `<tr class="${item.pagado ? 'pagado' : ''}">
                    <td><input type="checkbox" class="check-pago" ${item.pagado ? 'checked' : ''} onchange="toggle(${item.id})"></td>
                    <td>${item.fechaStr}</td><td>${item.desc}</td><td>Q${item.monto.toFixed(2)}</td>
                    <td><button class="btn-delete" onclick="borrar(${item.id})">×</button></td>
                </tr>`;
            });

            const interesCiclo = totalPendienteCiclo * TASA_MENSUAL;

            html += `</tbody></table>
                    <div class="resumen-footer">
                        <div class="resumen-item"><small>TOTAL PENDIENTE DEL MES</small><br><span class="monto-footer text-info-footer">Q${totalPendienteCiclo.toFixed(2)}</span></div>
                        <div class="resumen-item"><small>INTERÉS PENDIENTE DEL MES</small><br><span class="monto-footer text-danger-footer">Q${interesCiclo.toFixed(2)}</span></div>
                    </div>
                    <div class="banner-ultimo-dia">
                        Último día para pagar sin intereses: <b>${ciclo.limite}</b> | Pendiente: <b>Q${totalPendienteCiclo.toFixed(2)}</b>
                    </div>
                </div>`;
            cont.innerHTML += html;
        });

        const intGen = capPendTotal * TASA_MENSUAL;
        document.getElementById('soloCapital').innerText = capPendTotal.toFixed(2);
        document.getElementById('soloInteres').innerText = intGen.toFixed(2);
        document.getElementById('sumaPendienteGeneral').innerText = (capPendTotal + intGen).toFixed(2);
        document.getElementById('cajaGranTotal').style.display = todasLasCompras.length > 0 ? "block" : "none";
    }
</script>
